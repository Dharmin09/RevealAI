rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow anyone to create contact form submissions
    match /contactform/{document=**} {
      allow create: if request.resource.data.keys().hasAll(['name', 'email', 'message']);
      allow read: if request.auth != null;
    }
    
    // Allow authenticated users to read/write to their own data
    match /users/{userId} {
      allow read, write: if request.auth.uid == userId;
    }
    
    // Allow authenticated users to access analysis results
    match /analysis/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Default: deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
